<h1>install project </h1>
<pre>
<code>
    nvm install 18
</code>
</pre>
<pre>
<code>
    nvm use 18
</code>
</pre>
<pre>
<code>
    set NODE_OPTIONS=--openssl-legacy-provider
</code>
</pre>
<pre>
<code>
    npm start
</code>
</pre>

</pre>
<pre>
<code>json-server --watch db.json </code>
</pre>
